import cors from "cors";

app.use(express.json());

  token: "AWRsAAIncDIyODBjZjYyMzA4NWY0NjFlYmZiNDQzNzQ0MzI2MzkxM3AyMjU3MDg"

let clients = [];

app.post("/send", async (req, res) => {
  const { user, text } = req.body;

  const message = {
    user,
    text,
    time: Date.now()
  };

  await redis.rpush("chat:messages", JSON.stringify(message));

  // Also push to a notification queue for BLPOP
  await redis.rpush("chat:notifications", JSON.stringify(message));

  // Notify all waiting clients
  clients.forEach(c => {
    clearTimeout(c.timeout);
    c.res.json([message]);
  });

  clients = [];
  res.json({ success: true });
});

app.get("/poll", async (req, res) => {
  try {
    const lastTime = Number(req.query.lastTime || 0);

    // 1. Check existing messages first
    const messages = await redis.lrange("chat:messages", 0, -1);
    const parsed = messages.map(m =>
      typeof m === "string" ? JSON.parse(m) : m
    );

    const newMessages = parsed.filter(m => m.time > lastTime);

    if (newMessages.length > 0) {
      return res.json(newMessages);
    }

    // 2. If no messages, use BLPOP to wait for new ones
    const timeout = setTimeout(() => {
      // Clean up client from waiting list
      clients = clients.filter(c => c.res !== res);
      res.json([]);
    }, 25000); // 25 seconds timeout

    // Store client for traditional notification (fallback)
    const client = { res, timeout };
    clients.push(client);

    // Remove client if connection closes
    req.on("close", () => {
      clearTimeout(timeout);
      clients = clients.filter(c => c.res !== res);
    });

    // ========== BLPOP OPTIMIZATION ==========
    // Try to get a notification immediately using BLPOP
    // This is non-blocking for the main flow
    try {
      const notification = await redis.blpop("chat:notifications", 0.1); // Very short timeout
      
      if (notification && notification[1]) {
        const newMessage = JSON.parse(notification[1]);
        
        // Check if this message is new for this client
        if (newMessage.time > lastTime) {
          clearTimeout(timeout);
          clients = clients.filter(c => c.res !== res);
          return res.json([newMessage]);
        }
      }
    } catch (blpopErr) {
      // BLPOP failed, continue with traditional polling
      console.log("BLPOP check failed, using traditional polling");
    }

  } catch (err) {
    console.error("Poll error:", err);
    res.status(500).json({ error: "Polling failed" });
  }
});
